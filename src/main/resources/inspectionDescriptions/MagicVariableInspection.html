Reports the usage of magic variables in PHP files, which can lead to code that is difficult to maintain and understand.

The inspection helps detect occurrences of magic variables in your PHP codebase, promoting better coding practices by encouraging the use of explicit and well-defined variables instead.

It achieves this by analyzing PHP files and identifying any instances where magic variables are utilized. The inspection highlights these occurrences and suggests replacing them with explicitly defined variables. This can help improve code readability and maintainability, reducing the likelihood of bugs and enhancing overall code quality.

Embed code snippets: 


// automatically highlighted according to inspection registration 'language' attribute
package com.github.gamefixxer.phpstormconstantmanagerplugin

import com.intellij.codeInspection.LocalInspectionTool
import com.intellij.codeInspection.ProblemsHolder
import com.intellij.psi.PsiElementVisitor
import com.jetbrains.php.lang.psi.PhpFile

class MagicVariableInspection : LocalInspectionTool() {
    override fun buildVisitor(holder: ProblemsHolder, isOnTheFly: Boolean): PsiElementVisitor {
        return object : PsiElementVisitor() {
            override fun visitElement(element: com.intellij.psi.PsiElement) {
                if (element is PhpFile) {
                    element.accept(MagicVariableVisitor(holder.project))
                }
            }
        }
    }
}
